  \textcolor{#2}{\rule{#1}{0.5pt}}\nobreak}

\newcommand{\fabstract}[2]{%
\begin{center}
  {\huge \textcolor{\fabstitlecolour}{\textbf{#1}}}\\
  \vspace{-.17cm}
  \ulksection{.7\textwidth}{\fabstitlecolour}
  %\vspace{.1cm}
  ~\\
  {\normalsize
  \begin{minipage}{.68\textwidth}
    ~\\#2
  \end{minipage}}
\end{center}
}
